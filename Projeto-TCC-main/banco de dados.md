CREATE DATABASE SASC;
USE SASC;

CREATE TABLE GERENTE(
  ID INT NOT NULL PRIMARY KEY,
  NOME VARCHAR(50),
  CARGO VARCHAR(50),
  EMPRESA VARCHAR(100)
)

CREATE TABLE PRODUTO(
  ID INT NOT NULL PRIMARY KEY,
  NOME_PRODUTO VARCHAR(100),
  EM_ESTOQUE BOOLEAN,
  MEDIA_FEEDBACK REAL,
  QUANT_FEEDBACK INT
)

CREATE TABLE SETORES(
  ID INT NOT NULL PRIMARY KEY,
  AVALIACAO INT,
  SATISFACAO INT,
  RENDIMENTO INT,
  ATENDIMENTO INT
)

CREATE TABLE FEEDBACKS(
  ID INT NOT NULL PRIMARY  KEY,
  TOTAL INT,
  MEDIA_FEEDBACKS INT,
  COMENTARIO VARCHAR(300),
  DATA timestamp
)
